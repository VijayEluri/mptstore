65      public synchronized Set<Triple> getNextSet() {
66          _currentResource++;
67          Set<Triple> set = new HashSet<Triple>();
68          URIReference subject = getResource(_currentResource);
69          addRels(subject, set);
70          addPlains(subject, set);
71          addLocals(subject, set);
72          addLongs(subject, set);
73          addDoubles(subject, set);
74          addDateTimes(subject, set);

78      private void addRels(final SubjectNode subject, final Set<Triple> set) {
79          for (int i = 1; i <= _relsPerSubject; i++) {
80              set.add(new Triple(subject, getURIReference("urn:rel:" + i),
81                      getResource(_currentResource + i)));
82          }
83      }

85      private void addPlains(final SubjectNode subject, final Set<Triple> set) {
86          for (int i = 1; i <= _plainsPerSubject; i++) {
87              set.add(new Triple(subject, getURIReference("urn:plain:" + i),
88                      new Literal("plain " + i + " for resource " 
89                      + _currentResource)));
90          }
91      }

93      private void addLocals(final SubjectNode subject, final Set<Triple> set) {
94          for (int i = 1; i <= _localsPerSubject; i++) {
95              set.add(new Triple(subject, getURIReference("urn:local:" + i),
96                      getLocalLiteral("local for resource "
97                      + _currentResource, i)));

99      }

101     private void addLongs(final SubjectNode subject, final Set<Triple> set) {

113                     new Literal("" + _currentDouble, XSD_DOUBLE)));

117     private void addDateTimes(final SubjectNode subject,
