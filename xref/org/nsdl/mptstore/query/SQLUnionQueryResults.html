102         _sqlProvider = sqlProvider;
103         _fetchSize = fetchSize;
104         _autoReleaseConnection = autoReleaseConnection;

107

110             try {

122     /**
123      * Set _nextTuple to the next tuple.
124      *
125      * If there are no more tuples, proactively close and set
126      * <code>_nextTuple</code> to <code>null</code>.
127      *
128      * @throws QueryException if an unexpected error occurs.
129      */
130     private void readNextTuple() throws QueryException {
131 
132         try {
133             while (_results != null && !_results.next()) {
134                 startNextQuery();
135             }
136 
137             if (_results == null) {
138                 _nextTuple = null;
139             } else {
140                 int tupleSize = _sqlProvider.getTargets().size();
141                 _nextTuple = new ArrayList<Node>(tupleSize);
142                 for (int i = 1; i <= tupleSize; i++) {
143                     String nodeString = DBUtil.getLongString(_results, i);
144                     try {
145                         if (nodeString != null) {
146                             _nextTuple.add(NTriplesUtil.parseNode(nodeString));
147                         } else {
148                             _nextTuple.add(null);
149                         }
150                     } catch (ParseException e) {
151                         throw new QueryException("Error parsing RDF node ("
152                                 + nodeString + ") from database: "
153                                 + e.getMessage(), e);
154                     }
155                 }
156             }
157         } catch (SQLException e) {
158             throw new QueryException("Error querying database", e);
159         }
160     }

162     /**
163      * Start the next SQL query, setting _results as appropriate.
164      *
165      * Any resources tied up by the prior query, if any, will be cleaned up.
166      * If there are no more queries, _results will be set to <code>null</code>.
167      *
168      * @throws SQLException if there is a database error.
169      */
170     private void startNextQuery() throws SQLException {
171         if (_queries.hasNext()) {
172             if (_results != null) {
173                 // close previous query
174                 _results.close();
175                 _statement.close();
176             }
177             _statement = _conn.createStatement();
178             _statement.setFetchSize(_fetchSize);
179             _statement.setFetchDirection(ResultSet.FETCH_FORWARD);
180             String query = _queries.next();
181             LOG.info("Executing query:\n" + query);
182             _results = _statement.executeQuery(query);
183         } else {
184             LOG.info("Finished executing all queries");
185             close(); // proactively close if no more queries
186             _results = null;
187         }
188     }
