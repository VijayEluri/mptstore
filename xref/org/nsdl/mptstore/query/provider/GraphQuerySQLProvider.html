28  /**
29   * Translates a {@link GraphQuery} into a series of SQL statements.
30   * <p>
31   * Produces ANSI SQL-92 queries by converting each {@link GraphPattern} leaf of
32   * the query tree into a series of JOINs. Each join condition is formed by
33   * matching variables between {@link TriplePattern}s in the appripriate
34   * GraphPatterns.
35   * </p>
36   * <p>
37   * TODO:
38   * <ul>
39   * <li> Truly handle nested subqueries (currently is set to barf if a required
40   * or optional compomnent is not a GraphPattern). Things are set up for doing
41   * so, just not finished yet</li>
42   * <li> Implement a strategy for dealing with unbound predicates </li>
43   * </ul>
44   * </p>
45   *
46   * @author birkland
47   */
48  public class GraphQuerySQLProvider implements SQLBuilder, SQLProvider {

88      /**
89       * Choose the variables that define result tuples
90       * <p>
91       * The given list of variables are used for determining which bound values
92       * are included in result tuples, and in what order. If a variable is
93       * specified as a target, it <em>must</em> be present somewhere in the
94       * query. Any unmatched target will result in error
95       * </p>
96       *
97       * @param targetList
98       *            the list of query variables.
99       */
100     public void setTargets(final List<String> targetList) {
101         this.targets = new ArrayList<String>(targetList);
102

104     }

106     /**
107      * Force an order on the results.
108      * <p>
109      * Given the name of a target variable, results will be ordered by its bound
110      * value. Results may be specified to return in ascending or descending
111      * order.
112      * </p>
113      *
114      * @param target
115      *            Name of the target variable whose value will be the sort key.
116      * @param desc
117      *            True if results are to be in desceiding order, false
118      *            otherwise.
119      */
120     public void orderBy(final String target, final boolean desc) {
121         if (this.targets == null || !this.targets.contains(target)) {
122             throw new IllegalArgumentException("Cannot group by variable '"
123                     + target + "' since it is not in the target list "
124                     + targets);
125         }
126 
127         this.ordering = target;
128         if (desc) {
129             this.orderingDirection = "DESC";
130         } else {
131             this.orderingDirection = "ASC";
132         }
133     }

135     /**
136      * Returns a query in ANSI SQL
137      * <p>
138      * Translates the GraphQuery defined in the constructor, along with any
139      * specified orderings , into a set of SQL statements. The union of all SQL
140      * statement results, executed in order, will represent the entire result
141      * set.
142      * </p>
143      *
144      * @return list of SQL statements
145      * @throws QueryException
146      *             if there is some error translating the query to SQL.
147      */
148     public List<String> getSQL() throws QueryException {
149 
150         this.manager = new MappingManager(tableManager);
151         this.encounteredPatterns = new HashSet<MappableTriplePattern>();
152 
153         valueBindings = new HashMap<String, Set<String>>();
154 
155         /* These are bindings of variables in the 'required' query portions */
156         HashMap<String, String> requiredBindings =
157             new HashMap<String, String>();
158 
159         /* All variable bindings, including optional */
160         HashMap<String, String> allBindings = new HashMap<String, String>();
161 
162         /* This is the main sequence of joins constituting the query */
163         JoinSequence joinSeq = null;
164 
165         /* Process required elements first */
166         for (QueryElement e : query.getRequired()) {
167 
168             LOG.debug("Processing required element: " + e);
169 
170             /* Disallow subqueries for the time being */
171             if (e.getType().equals(QueryElement.Type.GraphQuery)) {
172                 throw new QueryException("Currently, we do not support "
173                         + "subqueries");
174             } else if (!e.getType().equals(QueryElement.Type.GraphPattern)) {
175                 /* Currently, this will never happen */
176                 throw new QueryException("Unknown query element type "
177                         + e.getType());
178             }
179 
180             /* All required elements of the query are inner joined */
181             if (joinSeq == null) {
182                 joinSeq = new JoinSequence(parseGraphPattern((GraphPattern) e,
183                         requiredBindings));
184             } else {
185                 joinSeq.addJoin(JoinType.INNER_JOIN, parseGraphPattern(
186                         (GraphPattern) e, requiredBindings), requiredBindings);
187             }
188         }
189 
190         allBindings.putAll(requiredBindings);
191 
192         for (QueryElement e : query.getOptional()) {
193 
194             LOG.debug("processing optional path: " + e);
195 
196             HashMap<String, String> optionalBindings = requiredBindings;
197 
198             /* Disallow subqueries for the time being */
199             if (e.getType().equals(QueryElement.Type.GraphQuery)) {
200                 throw new QueryException("Currently, we do not support "
201                         + "subqueries");
202             } else if (!e.getType().equals(QueryElement.Type.GraphPattern)) {
203                 /* Currently, this will never happen */
204                 throw new QueryException("Unknown query element type "
205                         + e.getType());
206             }
207 
208             joinSeq.addJoin(JoinType.LEFT_OUTER_JOIN, parseGraphPattern(
209                     (GraphPattern) e, optionalBindings), requiredBindings);
210 
211             addNewMappings(optionalBindings, allBindings);
212 
213         }
214 
215         StringBuilder sql = new StringBuilder();
216 
217         if (joinSeq == null) {
218             /* No tables to join means no query parts submitted */
219             return Arrays.asList("SELECT 1 WHERE 1=0");
220         } else {
221             sql.append("SELECT " + generateTargets(allBindings) + " FROM "
222                     + joinSeq);
223         }
224         /*
225          * If there are any values or constraints that remain to be added to the
226          * query, add them in a WHERE clause. NB: They better not be from an
227          * optional clause: It would probably be wise to either check here, or
228          * prove that an exception would have been thrown already.
229          */
230 
231         StringBuilder additional = new StringBuilder();
232         if (valueBindings.size() > 0) {
233             additional.append(" WHERE ");
234             ArrayList<String> valueKeys = new ArrayList<String>(valueBindings
235                     .keySet());
236             for (int i = 0; i < valueKeys.size(); i++) {
237                 ArrayList<String> values = new ArrayList<String>(valueBindings
238                         .get(valueKeys.get(i)));
239                 for (int j = 0; j < values.size(); j++) {
240                     if (i > 0 || j > 0) {
241                         sql.append(" AND ");
242                     }
243 
244                     LOG.debug("Adding remaining unused binding for "
245                             + valueKeys.get(i) + ", " + values.get(j) + "\n");
246                     additional.append(values.get(j));
247                 }
248             }
249         }
250 
251         if (!additional.toString().equals(" WHERE ")) {
252             sql.append(additional.toString());
253         }
254 
255         if (ordering != null) {
256             sql.append(" ORDER BY " + allBindings.get(ordering) + " "
257                     + orderingDirection);
258 
259         }
260 
261         ArrayList<String> sqlList = new ArrayList<String>();
262         sqlList.add(sql.toString());
263         return sqlList;
264     }

271     private Joinable parseGraphPattern(final GraphPattern g,
272             final HashMap<String, String> variableBindings)
273             throws QueryException {
274 
275         LOG.debug("parsing graph pattern " + g);
276         /* First, organize the filters by variable so that we can map them */
277         HashMap<String, Set<MappableNodeFilter>> filters =
278             new HashMap<String, Set<MappableNodeFilter>>();
279 
280         /* We're given NodeFilters, but need MappableNodeFilters. Convert. */
281         Set<MappableNodeFilter<Node>> givenFilters =
282             new HashSet<MappableNodeFilter<Node>>();
283         for (NodeFilter<Node> filter : g.getFilters()) {
284             givenFilters.add(new MappableNodeFilter<Node>(filter));
285         }
286 
287         /*
288          * For each given filter, if the value or constraint is a variable,
289          * place it in the filter pool.
290          */
291         for (MappableNodeFilter f : givenFilters) {
292             if (f.getNode().isVariable()) {
293                 if (!filters.containsKey(f.getNode().getVarName())) {
294                     LOG.debug("Adding " + f.getNode().getVarName()
295                             + " To filter pool..\n");
296                     filters.put(f.getNode().getVarName(),
297                             new HashSet<MappableNodeFilter>());
298                 }
299                 filters.get(f.getNode().getVarName()).add(f);
300             }
301             if (f.getConstraint().isVariable()) {
302                 if (!filters.containsKey(f.getConstraint().getVarName())) {
303                     LOG.debug("Adding " + f.getConstraint().getVarName()
304                             + " To filter pool..\n");
305                     filters.put(f.getConstraint().getVarName(),
306                             new HashSet<MappableNodeFilter>());
307                 }
308                 filters.get(f.getConstraint().getVarName()).add(f);
309             }
310 
311             if (!(f.getNode().isVariable() || f.getConstraint().isVariable())) {
312                 throw new IllegalArgumentException("Triple filters must "
313                         + "contain a variable.  Neither "
314                         + f.getNode().getVarName() + " nor "
315                         + f.getConstraint().getVarName() + " is a variable!");
316             }
317         }
318 
319         /* Next, process each triple pattern in this graph pattern */
320         LinkedList<MappableTriplePattern> steps =
321             new LinkedList<MappableTriplePattern>();
322         for (TriplePattern p : g.getTriplePatterns()) {
323             steps.add(new MappableTriplePattern(p));
324         }
325 
326         /*
327          * We create an initial join sequence from the first triple pattern NB:
328          * We require that all triple patterns in a graph pattern can be joined,
329          * so it is OK to do this
330          */
331 
332         MappableTriplePattern step = null;
333         boolean successfullyBoundFirstStep = false;
334         while (!steps.isEmpty()) {
335             step = steps.removeFirst();
336 
337             if (!bindPattern(step, variableBindings)) {
338                 continue;
339             } else {
340                 successfullyBoundFirstStep = true;
341                 break;
342             }
343         }
344 
345         if (!successfullyBoundFirstStep) {
346             LOG.info("Pattern is entirely redundant.  Ignoring: " + g);
347             return null;
348         }
349 
350         JoinSequence joins = new JoinSequence(new JoinTable(step));
351 
352         /* We keep track of all tje variables that are available to join on */
353         Set<MappableNodePattern> joinableVars = joins.joinVars();
354 
355         /* For all the remaining steps.. */
356         while (!steps.isEmpty()) {
357             step = getJoinablePattern(steps, variableBindings);
358             if (step == null) {
359                 throw new QueryException("Cannot bind all query steps! \n"
360                         + "remaining:\n" + steps
361                         + "\nvariables already bound:\n "
362                         + variableBindings.keySet() + "\n");
363             }
364             steps.remove(step);
365 
366             bindPattern(step, variableBindings);
367             JoinTable table = new JoinTable(step);
368 
369             joinableVars.addAll(table.joinVars());
370             JoinConditions conditions = new JoinConditions();
371 
372             /* Create all possible joins */
373             addJoinConditions(conditions, step, variableBindings);
374 
375             /* Add any filter constraints */
376             addFilterConditions(conditions, filters, joinableVars,
377                     variableBindings);
378 
379             /* Fold in any remaining constant bindings */
380             for (MappableNodePattern var : joinableVars) {
381                 if (valueBindings.containsKey(var.boundTable().alias())) {
382 
383                     for (String condition : valueBindings.get(var.boundTable()
384                             .alias())) {
385                         LOG.debug("parseGraphPattern: Adding remaining "
386                                 + "constant conditions " + condition + "\n");
387                         conditions.addCondition(condition);
388                     }
389                     valueBindings.remove(var.boundTable().alias());
390                 }
391             }
392 
393             /* Finally, add the join to the sequence */
394             joins.addJoin(JoinType.INNER_JOIN, table, conditions);
395         }
396 
397         /*
398          * If we still have filters yet unapplied, that is legitimate only if
399          * this pattern has a length of 1 AND it contains a variable that
400          * matches the filter...
401          */
402         if (filters.values().size() > 0 && g.getTriplePatterns().size() > 1) {
403             throw new QueryException("Filter is unbound");
404         }
405 
406         /* .. If so, then get that one pattern */
407         MappableTriplePattern p = new MappableTriplePattern(g
408                 .getTriplePatterns().get(0));
409 
410         /* ... and Process any remaining filters */
411         for (String varName : filters.keySet()) {
412             for (MappableNodeFilter f : filters.get(varName)) {
413                 processFilter(p, varName, f);
414             }
415         }
416         return joins;
417     }

419     private void addJoinConditions(final JoinConditions conditions,
420             final MappableTriplePattern step,
421             final HashMap<String, String> variableBindings) {
422         for (MappableNodePattern p : step.getNodes()) {
423             if (isBound(p, variableBindings)) {
424                 /* Join this var's column w/the corresponding bound one */
425                 if (!p.mappedName().equals(getBoundValue(
426                         p, variableBindings))) {
427                     LOG.debug("parseGraphPattern: Adding Join Condition "
428                             + p.mappedName() + " = "
429                             + getBoundValue(p, variableBindings) + "\n");
430                     conditions.addCondition(p.mappedName(), " = ",
431                             getBoundValue(p, variableBindings));
432 
433                     if (valueBindings.containsKey(p.boundTable().alias())) {
434                         LOG.debug("Removing value binding from queue: "
435                                 + p.mappedName() + " = "
436                                 + getBoundValue(p, variableBindings) + "\n");
437                         valueBindings.get(p.boundTable().alias()).remove(
438                                 p.mappedName() + " = "
439                                         + getBoundValue(p, variableBindings));
440                     }
441                 }
442             }
443         }
444     }

446     private void addFilterConditions(final JoinConditions conditions,
447             final HashMap<String, Set<MappableNodeFilter>> filters,
448             final Set<MappableNodePattern> joinableVars,
449             final HashMap<String, String> variableBindings) {
450         for (String filterVar : new ArrayList<String>(filters.keySet())) {
451             for (MappableNodePattern joinableVar : joinableVars) {
452                 if (joinableVar.isVariable()
453                         && joinableVar.getVarName().equals(filterVar)) {
454                     for (MappableNodeFilter f : filters.get(filterVar)) {
455                         String right;
456                         String left;
457 
458                         if (f.getNode().isVariable()
459                                 && f.getNode().getVarName().equals(filterVar)) {
460                             left = getBoundValue(joinableVar, variableBindings);
461                         } else if (f.getNode().isVariable()) {
462                             left = getBoundValue(f.getNode(), variableBindings);
463                         } else {
464                             left = DBUtil.quotedString(f.getNode().getNode()
465                                     .toString(), backslashEscape);
466                         }
467 
468                         if (f.getConstraint().isVariable()
469                                 && f.getConstraint().getVarName().equals(
470                                         filterVar)) {
471                             right = getBoundValue(
472                                     joinableVar, variableBindings);
473                         } else if (f.getConstraint().isVariable()) {
474                             right = getBoundValue(f.getConstraint(),
475                                     variableBindings);
476                         } else {
477                             right = DBUtil.quotedString(f.getConstraint()
478                                     .getNode().toString(), backslashEscape);
479                         }
480 
481                         conditions.addCondition(left, f.getOperator(), right);
482                         LOG.debug("parseGraphPattern: Adding filter "
483                                 + "condition: " + left + " " + f.getOperator()
484                                 + " " + right + "\n");
485                     }
486 
487                     removeFromMap(filters.get(filterVar), filters);
488                 }
489             }
490         }
491     }

493     private void processFilter(final MappableTriplePattern p,
494             final String varName, final MappableNodeFilter f)
495             throws QueryException {
496 
497         String mappedName;
498         if (p.getSubject().isVariable()
499                 && p.getSubject().getVarName().equals(varName)) {
500             mappedName = p.getSubject().mappedName();
501         } else if (p.getObject().isVariable()
502                 && p.getObject().getVarName().equals(varName)) {
503             mappedName = p.getObject().mappedName();
504         } else {
505             throw new QueryException("Variable " + varName
506                     + " in filter Cannot be found in graph query");
507         }
508 
509         if (!valueBindings.containsKey(mappedName)) {
510             valueBindings.put(mappedName, new HashSet<String>());
511         }
512 
513         if (f.getNode().isVariable()
514                 && f.getNode().getVarName().equals(varName)) {
515             if (f.getConstraint().isVariable()) {
516                 /* XXX It's probably not legal to be here.. */
517                 LOG.warn("Node filter constraint is variable?  "
518                         + "It's probably not legal to be here...");
519             } else {
520                 valueBindings.get(mappedName)
521                         .add(
522                                 mappedName
523                                         + " "
524                                         + f.getOperator()
525                                         + " "
526                                         + DBUtil.quotedString(f.getConstraint()
527                                                 .getNode().toString(),
528                                                 backslashEscape));
529                 LOG.debug("Remaining Filters: " + mappedName + " "
530                         + f.getOperator() + " '" + f.getConstraint().getNode()
531                         + "'" + "\n");
532             }
533         } else if (f.getConstraint().isVariable()
534                 && f.getConstraint().getVarName().equals(varName)) {
535             if (f.getNode().isVariable()) {
536                 /* XXX: it's probably not legal to be here.. */
537                 LOG.warn("Node filter constraint is variable and node is "
538                         + "variable?  It's probably not legal to be here...");
539             } else {
540                 valueBindings.get(mappedName).add(
541                         DBUtil.quotedString(f.getNode().getNode().toString(),
542                                 backslashEscape)
543                                 + " " + f.getOperator() + " " + mappedName);
544                 LOG.debug("Remaining Filters: " + "'" + f.getNode().getNode()
545                         + "' " + f.getOperator() + " " + mappedName + "\n");
546             }
547         }
548 
549     }

551     /*
552      * From a list of mappable triple patterns, pick one that has a variable
553      * that also occurs in variablebindings. This assures that if the caller is
554      * looking for joins, it will find at least one.
555      */
556     private MappableTriplePattern getJoinablePattern(
557             final List<MappableTriplePattern> l,
558             final HashMap<String, String> variableBindings) {
559         for (MappableTriplePattern p : l) {
560             if (isBound(p.getSubject(), variableBindings)
561                     || isBound(p.getObject(), variableBindings)) {
562                 return p;
563             }
564         }
565         return null;
566     }

568     /*
569      * Determine if a variable has been apped to a literal or specific column of
570      * a table
571      */
572     private boolean isBound(final MappableNodePattern n,
573             final HashMap<String, String> variableBindings) {
574         if (n.isVariable()) {
575             return variableBindings.containsKey(n.getVarName());
576         } else {
577             return true;
578         }
579     }

581     /*
582      * Get either an explicit value or a column/table reference for a given
583      * mapped node pattern
584      */
585     private String getBoundValue(final MappableNodePattern n,
586             final HashMap<String, String> variableBindings) {
587         if (n.isVariable()) {
588             return variableBindings.get(n.getVarName());
589         } else {
590             return DBUtil.quotedString(n.getNode().toString(), backslashEscape);
591         }
592     }

594     /*
595      * Bind the variables/values of a triple pattern by: - Placing any new
596      * variables into the master bings map, - Placing any literal values into
597      * the literals map
598      */
599     private boolean bindPattern(final MappableTriplePattern t,
600             final HashMap<String, String> variableBindings) {
601 
602         if (!encounteredPatterns.contains(t)) {
603             encounteredPatterns.add(t);
604             LOG.debug("Processing new pattern " + t);
605             t.bindTo(manager.mapPredicateTable(t.getPredicate()));
606 
607             bindNode(t.getSubject(), variableBindings);
608             // bindNode(t.predicate, variableBindings);
609             bindNode(t.getObject(), variableBindings);
610             return true;
611         } else {
612             LOG.info("Already encountered pattern " + t);
613             return false;
614         }
615     }

617     /* TODO: be able to bind a predicate node */
618     private void bindNode(final MappableNodePattern p,
619             final HashMap<String, String> variableBindings) {
620         if (p.isVariable()) {
621             StringBuilder existing = new StringBuilder();
622             for (String variable : variableBindings.keySet()) {
623                 existing.append(variable + " = "
624                         + variableBindings.get(variable) + "\n");
625             }
626             LOG.debug("Considering " + p.getVarName()
627                     + " with respect to \n" + existing);
628             if (!variableBindings.containsKey(p.getVarName())) {
629                 LOG.debug("Bound " + p.getVarName() + " to " + p.mappedName()
630                         + "\n");
631                 variableBindings.put(p.getVarName(), p.mappedName());
632             }
633         } else {
634             if (!valueBindings.containsKey(p.boundTable().alias())) {
635                 valueBindings
636                         .put(p.boundTable().alias(), new HashSet<String>());
637             }
638             LOG.debug("bindNode: adding valueBinding " + p.mappedName() + " = "
639                     + "'" + p.getNode() + "'\n");
640             valueBindings.get(p.boundTable().alias()).add(
641                     p.mappedName()
642                             + " = "
643                             + DBUtil.quotedString(p.getNode().toString(),
644                                     backslashEscape));
645         }
646     }

648     /**
649      * Removes a mapped value (and all associated keys) from a map.
650      *
651      * @param value
652      *            mapped value to remove
653      * @param m
654      *            map to remove the value from
655      */
656     private <K, V> void removeFromMap(final V value, final Map<K, V> m) {
657         Set<K> toRemove = new HashSet<K>();
658         for (Map.Entry<K, V> e : m.entrySet()) {
659             if (e.getValue().equals(value)) {
660                 toRemove.add(e.getKey());
661             }
662         }
663 
664         for (K key : toRemove) {
665             m.remove(key);
666         }
667     }

669     private String generateTargets(
670             final HashMap<String, String> variableBindings) {
671         String selects = "";
672         for (int i = 0; i < targets.size(); i++) {
673             selects += variableBindings.get(targets.get(i));
674             if (i < targets.size() - 1) {
675                 selects += ", ";
676             }
677         }
678         return selects;
679     }

681     private <K, V> void addNewMappings(
682             final Map<K, V> from, final Map<K, V> to) {
683         for (K key : from.keySet()) {
684             if (!to.containsKey(key)) {
685                 to.put(key, from.get(key));
686             }
687         }

738         private final StringBuilder join;
739 
740         private int joinCount = 0;
741 
742         private final List<Joinable> joined = new ArrayList<Joinable>();
743 
744         public JoinSequence(final Joinable start) {
745             this.join = new StringBuilder(start.declaration());
746             joined.add(start);
747             joinCount = 1;
748         }
749 
750         public void addJoin(final String joinType, final Joinable j,
751                 final String joinConstraints) {
752             if (j == null) {
753                 LOG.info("Skipping join");
754                 return;
755             }
756             join.append(" " + joinType + " " + j.declaration());
757             joined.add(j);
758             if (joinConstraints != null && joinConstraints != "") {
759                 join.append(" ON (" + joinConstraints + ")");
760             }
761             joinCount++;
762         }

764         public void addJoin(final String joinType, final Joinable j,
765                 final JoinConditions conditions) {
766             addJoin(joinType, j, conditions.toString());
767         }
768 
769         public void addJoin(final String joinType, final Joinable j,
770                 final HashMap<String, String> variableBindings) {
771 
772             if (j == null) {
773                 LOG.info("Skipping join");
774                 return;
775             }
776             JoinConditions conditions = new JoinConditions();
777 
778             for (MappableNodePattern existingVar : this.joinVars()) {
779                 for (MappableNodePattern candidateVar : j.joinVars()) {
780 
781                     String existingVarName = existingVar.getVarName();
782                     String candidateVarName = candidateVar.getVarName();
783                     String candidateBinding = variableBindings.get(candidateVar
784                             .getVarName());
785                     String existingBinding = existingVar.mappedName();
786 
787                     if (existingVarName.equals(candidateVarName)
788                             && existingBinding.equals(candidateBinding)) {
789                         conditions.addCondition(existingBinding, " = ",
790                                 candidateVar.mappedName());
791                     }
792                 }
793             }
794 
795             addJoin(joinType, j, conditions);
796         }

799             HashSet<MappableNodePattern> joinVars =

801

803                 joinVars.addAll(joinable.joinVars());
804             }
805 
806             return joinVars;
807         }

810             if (joinCount == 1) {
811                 return join.toString();
812             } else {
813                 return "(" + join.toString() + ")";
814             }
815         }

859     private class MappingManager {
860         private HashMap<String, List<String>> predicateMap =
861             new HashMap<String, List<String>>();
862 
863         private HashMap<PredicateNode, MPTable> nonexistantMappings =
864             new HashMap<PredicateNode, MPTable>();
865 
866         private final TableManager adaptor;
867 
868         private int allMap = 0;
869 
870         public MappingManager(final TableManager mgr) {
871             this.adaptor = mgr;
872         }
873 
874         public MPTable mapPredicateTable(
875                 final MappableNodePattern<PredicateNode> predicate) {
876             if (predicate.isVariable()) {
877                 throw new IllegalArgumentException("predicate must not "
878                         + "be a variable");
879             }
880 
881             String tableName;
882             String alias;
883 
884             if (predicate.isVariable()) {
885                 tableName = allTableQuery();
886                 alias = "ap_" + ++allMap;
887             } else {
888                 tableName = adaptor.getTableFor(predicate.getNode());
889             }
890 
891             if (tableName == null) {
892                 /* No predicate found.. create table that returns no results */
893                 alias = "np_" + nonexistantMappings.size();
894                 tableName = "(SELECT p AS s, p AS o from tmap where 1=0)";
895                 if (!nonexistantMappings.containsKey(predicate.getNode())) {
896                     alias = "np_" + nonexistantMappings.size();
897                     LOG.debug("No table for '" + predicate.getNode()
898                             + "'.  Using empty table as " + alias);
899                     tableName = "(SELECT p AS s, p AS o from tmap where 1=0)";
900                     nonexistantMappings.put(predicate.getNode(), new MPTable(
901                             tableName, alias));
902                     predicateMap.put(predicate.getNode().toString(),
903                             new ArrayList<String>());
904                     return nonexistantMappings.get(predicate.getNode());
905                 } else {
906                     List<String> aliases = predicateMap.get(predicate.getNode()
907                             .toString());
908                     String primaryAlias = nonexistantMappings.get(
909                             predicate.getNode()).alias();
910                     alias = primaryAlias + "_" + aliases.size();
911                     LOG.debug("Nonexistant predicate already encountered.  "
912                             + " Using alias " + alias);
913                     return new MPTable(tableName, alias);
914                 }
915             } else if (predicateMap.containsKey(
916                     predicate.getNode().toString())) {
917                 LOG.debug("Predicate already encountered.  "
918                         + "Making new table alias");
919                 List<String> aliases = predicateMap.get(predicate.getNode()
920                         .toString());
921                 alias = tableName + "_" + aliases.size();
922                 aliases.add(alias);
923             } else {
924                 LOG.debug("Predicate never encountered.  "
925                         + "Will refer to it by its own name");
926                 ArrayList<String> aliases = new ArrayList<String>();
927                 aliases.add(tableName);
928                 predicateMap.put(predicate.getNode().toString(), aliases);
929                 alias = tableName;
930             }
931 
932             LOG.debug("Mapping predicate " + predicate.getNode().getValue()
933                     + " to " + tableName + " as " + alias);
934             MPTable table = new MPTable(tableName, alias);
935             return table;
936         }

938         private String allTableQuery() {
939             boolean first = true;
940             StringBuilder allTable = new StringBuilder();
941             allTable.append("( ");
942             for (PredicateNode predicate : adaptor.getPredicates()) {
943                 if (first) {
944                     first = false;
945                 } else {
946                     allTable.append(" UNION ALL ");
947                 }
948 
949                 allTable.append(" SELECT * from "
950                         + adaptor.getTableFor(predicate));
951             }
952 
953             allTable.append(")");
954             return allTable.toString();
955         }
