100     /**
101      * Execute the given update operation of the given triples.
102      *
103      * @param conn The connection to execute the update on.
104      * @param triples The triples to add or delete.
105      * @param delete Boolean indicating whether the operation is an add
106      *        or delete.
107      * @throws ModificationException if the operation fails for any reason.
108      */
109     private void updateTriples(final Connection conn,
110                                final Iterator<Triple> triples,
111                                final boolean delete)
112             throws ModificationException {
113 
114         Map<PredicateNode, PreparedStatement> statements =
115                 new HashMap<PredicateNode, PreparedStatement>();
116 
117         try {
118             while (triples.hasNext()) {
119 
120                 Triple triple = triples.next();
121 
122                 if (LOG.isDebugEnabled()) {
123                     String prefix;
124                     if (delete) {
125                         prefix = "Deleting ";
126                     } else {
127                         prefix = "Adding ";
128                     }
129                     LOG.debug(prefix + triple.toString());
130                 }
131 
132                 PredicateNode predicate = triple.getPredicate();
133 
134                 PreparedStatement statement = statements.get(predicate);
135                 if (statement == null) {
136                     String table = _tableManager.getOrMapTableFor(predicate);
137                     String sql;
138                     if (delete) {
139                         sql = "DELETE FROM " + table
140                                 + " WHERE s = ? AND o = ?";
141                     } else {
142                         sql = "INSERT INTO " + table
143                                 + " (s, o) VALUES (?, ?)";
144                     }
145                     statement = conn.prepareStatement(sql);
146                     statements.put(predicate, statement);
147                 }
148 
149                 statement.setString(1, triple.getSubject().toString());
150                 statement.setString(2, triple.getObject().toString());
151                 statement.execute();
152             }
153         } catch (SQLException e) {
154             throw new ModificationException("Database update failed", e);
155         } finally {
156 
157             // close all statements we created for this update
158             Iterator<PreparedStatement> iter = statements.values().iterator();
159             while (iter.hasNext()) {
160                 PreparedStatement statement = iter.next();
161                 try {
162                     statement.close();
163                 } catch (SQLException e) {
164                     LOG.warn("unable to close statement", e);
165                 }
166             }
167         }
168     }

170     /** {@inheritDoc} */
171     public void deleteAllTriples(final Connection conn)
172             throws ModificationException {
173         try {
174             _tableManager.dropAllPredicateTables();
175         } catch (SQLException e) {
176             throw new ModificationException("Failed to delete all triples", e);
177         }

181     /** {@inheritDoc} */
182     public QueryResults query(final Connection connection,
183                               final QueryLanguage language,
184                               final int fetchSize,
185                               final boolean autoReleaseConnection,
186                               final String query)
187             throws QueryException {
188         QueryResults results = null;
189         try {
190             QueryCompiler compiler = _compilerMap.get(language);
191             if (compiler != null) {
192                 SQLProvider provider = compiler.compile(query);
193                 results = new SQLUnionQueryResults(connection,
194                                                    provider,
195                                                    fetchSize,
196                                                    autoReleaseConnection);
197                 return results;
198             } else {
199                 throw new QueryException("Query language not supported: "
200                         + language.getName());
201             }
202         } finally {
203             if (results == null && autoReleaseConnection) {
204                 DBUtil.release(connection);
205             }
206         }
207     }
