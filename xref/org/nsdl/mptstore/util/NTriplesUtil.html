78      /**
79       * Consume the next URI reference.
80       *
81       * This will advance the reader through the next '&gt;' character.
82       *
83       * @param reader the reader to get characters from.
84       * @param pos the current position in the overall input.
85       * @return all characters up to an including the terminal, '&gt;'.
86       * @throws IOException if there's an I/O error reading the input.
87       * @throws ParseException if the reader is exhausted before the terminal
88       *         is encountered.
89       */
90      public static String consumeURIReference(final Reader reader,
91                                               final int pos)
92              throws IOException, ParseException {
93  
94          int i = 0;
95          StringBuffer buf = new StringBuffer();
96  
97          int c = reader.read();
98          while (c != '>') {
99              if (c == -1) {
100                 throw new ParseException(EXPECTED_G, pos + i);
101             }
102             buf.append((char) c);
103             i++;
104             c = reader.read();
105         }
106         buf.append((char) c);
107         return buf.toString();
108     }

110     /**
111      * Tell whether the given character matches one in the given array.
112      *
113      * @param chars the array.
114      * @parm c the character.
115      * @return true if c exists in chars.
116      */
117     private static boolean isOneOf(final int[] chars,
118                                    final int c) {
119         for (int i = 0; i < chars.length; i++) {
120             if (c == chars[i]) {
121                 return true;
122             }
123         }
124         return false;
125     }

127     /**
128      * Move the reader to the first non-whitespace character
129      * and return the number of whitespaces encountered.
130      *
131      * @param reader the reader.
132      * @param pos the current position in the reader.
133      * @return the number of whitespaces encountered.
134      * @throws IOException if there is an error reading.
135      * @throws ParseException if the first character is not a whitespace or
136      *         EOF is reached before a non-whitespace char.
137      */
138     public static int consumeWhitespace(final Reader reader,
139                                         final int pos)
140             throws IOException, ParseException {
141 
142         int i = 0;
143 
144         int c = reader.read();
145         if (!isOneOf(SPACE_OR_TAB, c)) {
146             throw new ParseException(EXPECTED_ST, pos + i);
147         }
148         while (isOneOf(SPACE_OR_TAB, c)) {
149             if (c == -1) {
150                 throw new ParseException(UNEXPECTED_EOF, pos + i - 1);
151             }
152             i++;
153             reader.mark(Integer.MAX_VALUE);
154             c = reader.read();
155         }
156         reader.reset();
157         return i;
158     }

160     /**
161      * Extract the object part of a triple string.
162      *
163      * @param ntTriple the full triple string.
164      * @param i the position of the first object character.
165      * @throws ParseException if the input does not appear to be a valid
166      *         triple string.
167      */
168     private static String getObjectString(final String ntTriple,
169                                           final int i)
170             throws ParseException {
171 
172         int j = ntTriple.length() - 1;
173 
174         char ch = ntTriple.charAt(j);
175         while (ch != '.') {
176             if (ch != '\t' && ch != ' ') {
177                 throw new ParseException(EXPECTED_PST, j);
178             }
179             j--;
180             if (j < i) {
181                 throw new ParseException(EXPECTED_QLST, j);
182             }
183             ch = ntTriple.charAt(j);
184         }
185         j--;
186         if (j < i) {
187             throw new ParseException(EXPECTED_QLST, j);
188         }
189         ch = ntTriple.charAt(j);
190         while (ch == ' ' || ch == '\t') {
191             j--;
192             if (j < i) {
193                 throw new ParseException(EXPECTED_QLST, j);
194             }
195             ch = ntTriple.charAt(j);
196         }
197 
198         return ntTriple.substring(i, j + 1);
199     }

201     /**
202      * Parse an RDF triple in N-Triples format.
203      *
204      * @param ntTriple the input string.
205      * @return the parsed triple.
206      * @throws ParseException if the input syntax is incorrect
207      * @see <a href="http://www.w3.org/TR/rdf-testcases/#triple">
208      *        N-Triples triple syntax</a>
209      */
210     public static Triple parseTriple(final String ntTriple)
211             throws ParseException {
212 
213         StringReader reader = new StringReader(ntTriple);
214 
215         try {
216             int i = 0;
217 
218             // subject
219             String sString = consumeURIReference(reader, i);
220             SubjectNode subject = parseSubject(sString);
221             i += sString.length();
222 
223             // whitespace+
224             i += consumeWhitespace(reader, i);
225 
226             // predicate
227             String pString = consumeURIReference(reader, i);
228             PredicateNode predicate;
229             try {
230                 predicate = parsePredicate(pString);
231             } catch (ParseException e) {
232                 throw new ParseException(e.getMessage(),
233                         e.getErrorOffset() + i);
234             }
235             i += pString.length();
236 
237             // whitespace+
238             i += consumeWhitespace(reader, i);
239 
240             // object
241             String oString = getObjectString(ntTriple, i);
242             ObjectNode object;
243             try {
244                 object = parseObject(oString);
245             } catch (ParseException e) {
246                 throw new ParseException(e.getMessage(),
247                         e.getErrorOffset() + i);
248             }
249 
250             // triple
251             return new Triple(subject, predicate, object);
252 
253         } catch (IOException e) {
254             // should not happen -- we're using a StringReader
255             throw new RuntimeException("Unexpected IO error", e);
256         }
257 
258     }

326     /**
327      * Advance the reader until a double quote character is encountered
328      * and return the characters seen.
329      *
330      * @param reader the reader.
331      * @param pos the current position in the overall string.
332      * @throws IOException if there is an error reading.
333      * @throws ParseException if EOF is encountered before the terminal
334      *         character or an unescaped linefeed, carriage return,
335      *         or tab character is encountered.
336      */
337     private static String consumeLiteralValue(final Reader reader,
338                                               final int pos)
339             throws IOException, ParseException {
340 
341         StringBuffer escaped = new StringBuffer();
342         int i = 0;
343         int c = reader.read();
344 
345         while (c != '"') {
346 
347             if (c == -1) {
348                 throw new ParseException(EXPECTED_Q, pos + i);
349             }
350             escaped.append((char) c);
351 
352             if (c == '\\') {
353                 c = reader.read();
354                 i++;
355                 if (c == -1) {
356                     throw new ParseException(EXPECTED_Q, pos + i);
357                 }
358                 escaped.append((char) c);
359             } else if (c == '\r') {
360                 throw new ParseException(UNESCAPED_LF, pos + i);
361             } else if (c == '\n') {
362                 throw new ParseException(UNESCAPED_CR, pos + i);
363             } else if (c == '\t') {
364                 throw new ParseException(UNESCAPED_TAB, pos + i);
365             }
366 
367             c = reader.read();
368             i++;
369         }
370 
371         return escaped.toString();
372     }

374     /**
375      * Parse an RDF literal in N-Triples format.
376      *
377      * @param s the input string.
378      * @return the parsed literal.
379      * @throws ParseException if the input syntax is incorrect
380      */
381     public static Literal parseLiteral(final String s)
382             throws ParseException {
383 
384         StringReader reader = new StringReader(s);
385 
386         try {
387 
388             int first = reader.read();
389 
390             if (first != '"') {
391                 throw new ParseException(EXPECTED_Q, 0);
392             }
393 
394             int i = 1;
395 
396             String escaped = consumeLiteralValue(reader, i);
397             String value;
398             try {
399                 value = unescapeLiteralValue(escaped);
400             } catch (ParseException e) {
401                 throw new ParseException(e.getMessage(),
402                                          e.getErrorOffset() + i);
403             }
404             i += escaped.length();
405 
406             // c == '"', read next char
407             int c = reader.read();
408             i++;
409 
410             if (c == '@') {
411                 try {
412                     return new Literal(value, s.substring(i + 1));
413                 } catch (ParseException e) {
414                     throw new ParseException(e.getMessage(),
415                                              e.getErrorOffset() + i + 1);
416                 }
417             } else if (c == '^') {
418                 c = reader.read(); i++;
419                 if (c != '^') {
420                     throw new ParseException(EXPECTED_C, i);
421                 }
422                 try {
423                     URIReference datatype = parseURIReference(
424                             s.substring(i + 1));
425                     return new Literal(value, datatype);
426                 } catch (ParseException e) {
427                     throw new ParseException(e.getMessage(),
428                                              e.getErrorOffset() + i);
429                 }
430             } else if (c == -1) {
431                 return new Literal(value);
432             } else {
433                 throw new ParseException(EXPECTED_ACE, i);
434             }
435 
436         } catch (IOException e) {
437             // should not happen -- we're using a StringReader
438             throw new RuntimeException("Unexpected IO error", e);
439         }
440     }

443      * Parse an RDF URI reference in N-Triples format.
444      *
445      * @param s the input string.
446      * @return the parsed URI reference.
447      * @throws ParseException if the input syntax is incorrect
448      */
449     public static URIReference parseURIReference(final String s)
450             throws ParseException {
451 
452         char first = s.charAt(0);
453 
454         if (first != '<') {
455             throw new ParseException(EXPECTED_L, 0);
456         }
457 
458         char last = s.charAt(s.length() - 1);
459         if (last != '>') {
460             throw new ParseException(EXPECTED_G, s.length() - 1);
461         }
462 
463         try {
464             return new URIReference(s.substring(1, s.length() - 1));
465         } catch (URISyntaxException e) {
466             throw new ParseException(EXPECTED_ABS_URI, 1);
467         }
468     }

485     /**
486      * Return the next unescaped value from the escaped string.
487      *
488      * @param s the whole string.
489      * @param backslashPos the position of the current escape character.
490      * @param i the current position in the string.
491      * @param c the current character.
492      * @param buf the buffer to append to.
493      * @throws ParseException if an incomplete or illegal escape sequence
494      *         is encountered.
495      */
496     private static int appendUnescaped(final String s,
497                                        final int backslashPos,
498                                        final int i,
499                                        final char c,
500                                        final StringBuffer buf)
501             throws ParseException {
502 
503 
504         if (c == 'u') {
505             if (backslashPos + SHORT_ESCAPE_LENGTH >= s.length()) {
506                 throw new ParseException(INCOMPLETE_ESCAPE, i);
507             }
508             String xx = s.substring(backslashPos + 2,
509                     backslashPos + SHORT_ESCAPE_LENGTH + 1);
510             try {
511                 int nextChar = (char) Integer.parseInt(xx, HEX);
512                 buf.append((char) nextChar);
513                 return backslashPos + SHORT_ESCAPE_LENGTH + 1;
514             } catch (NumberFormatException e) {
515                 throw new ParseException(ILLEGAL_ESCAPE, i);
516             }
517         } else if (c == 'U') {
518             if (backslashPos + LONG_ESCAPE_LENGTH - 1 >= s.length()) {
519                 throw new ParseException(INCOMPLETE_ESCAPE, i);
520             }
521             String xx = s.substring(backslashPos + 2,
522                     backslashPos + LONG_ESCAPE_LENGTH);
523             try {
524                 int nextChar = (char) Integer.parseInt(xx, HEX);
525                 buf.append((char) nextChar);
526                 return backslashPos + LONG_ESCAPE_LENGTH;
527             } catch (NumberFormatException e) {
528                 throw new ParseException(ILLEGAL_ESCAPE, i);
529             }
530         } else {
531             buf.append(getUnescaped(c, i));
532             return backslashPos + 2;
533         }
534 
535     }

537     /**
538      * Get the unescaped value for the given character.
539      *
540      * The character was encountered in the string after an escape
541      * character.
542      *
543      * @param c the character.
544      * @param i the current position in the overall string.
545      * @throws ParseException if the given character does not
546      *         represent a single character found in an escape sequence.
547      */
548     private static char getUnescaped(final char c,
549                                      final int i) throws ParseException {
550         if (c == 't') {
551             return '\t';
552         } else if (c == 'r') {
553             return '\r';
554         } else if (c == 'n') {
555             return '\n';
556         } else if (c == '"' || c == '\\') {
557             return c;
558         } else {
559             throw new ParseException(UNESCAPED_BACKSLASH, i);
560         }
561     }

563     /**
564      * Unescape an N-Triples-escaped string.
565      *
566      * <ul>
567      *   <li> All input characters are validated to be 7-bit ASCII.</li>
568      *   <li> Unicode escapes (&#x5C;uxxxx and &#x5C;Uxxxxxxxx) are validated
569      *        to be complete and legal, and are restored to the value indicated
570      *        by the hexadecimal argument.</li>
571      *   <li> Backslash-escaped values (\t, \r, \n, \", and \\) are restored
572      *        to their original form (tab, carriage return, linefeed, quote,
573      *        and backslash, respectively).</li>
574      * </ul>
575      *
576      * @param s The input string.
577      * @return The unescaped string.
578      * @throws ParseException if the input syntax is incorrect
579      */
580     public static String unescapeLiteralValue(final String s)
581             throws ParseException {
582 
583         verifyAscii(s);
584 
585         int backslashPos = s.indexOf('\\');
586 
587         // return early if no escape
588         if (backslashPos == -1) {
589             return s;
590         }
591 
592         int i = 0;
593         StringBuffer buf = new StringBuffer(s.length());
594 
595         // unescape all
596         while (backslashPos != -1) {
597 
598             buf.append(s.substring(i, backslashPos));
599 
600             if (backslashPos + 1 >= s.length()) {
601                 throw new ParseException(UNESCAPED_BACKSLASH, i);
602             }
603 
604             char c = s.charAt(backslashPos + 1);
605             i = appendUnescaped(s, backslashPos, i, c, buf);
606 
607             backslashPos = s.indexOf('\\', i);
608         }
609         buf.append(s.substring(i));
610 
611         return buf.toString();
612     }

614     /**
615      * Escape a string to N-Triples literal format.
616      *
617      * <ul>
618      *   <li> Unicode escaping (&#x5C;uxxxx or &#x5C;Uxxxxxxxx, a
619      *        appropriate) will be used for all characters in the
620      *        following ranges:
621      *        <ul>
622      *          <li> 0x0 through 0x8</li>
623      *          <li> 0xB through 0xC</li>
624      *          <li> 0xE through 0x1F</li>
625      *          <li> 0x7F through 0xFFFF</li>
626      *          <li> 0x10000 through 0x10FFFF</li>
627      *        </ul>
628      *   <li> Backslash escaping will be used for double quote (\"),
629      *        backslash (\\), line feed (\n), carriage return (\r),
630      *        and tab (\t) characters.</li>
631      *   <li> All other characters will be represented as-is.</li>
632      * </ul>
633      *
634      * @param s The input string.
635      * @return The escaped string.
636      */
637     public static String escapeLiteralValue(final String s) {
638 
639         int len = s.length();
640         StringBuffer out = new StringBuffer(len * 2);
641 
642         for (int i = 0; i < len; i++) {
643             char c = s.charAt(i);
644             int cNum = (int) c;
645             if (c == '\\') {
646                 out.append("\\\\");
647             } else if (c == '"') {
648                 out.append("\\\"");
649             } else if (c == '\n') {
650                 out.append("\\n");
651             } else if (c == '\r') {
652                 out.append("\\r");
653             } else if (c == '\t') {
654                 out.append("\\t");
655             } else if (isLowUnicode(cNum)) {
656                 out.append("\\u");
657                 out.append(hexString(cNum, SHORT_ESCAPE_LENGTH - 1));
658             } else if (isHighUnicode(cNum)) {
659                 out.append("\\U");
660                 out.append(hexString(cNum, LONG_ESCAPE_LENGTH - 2));
661             } else {
662                 out.append(c);
663             }
664         }
665 
666         return out.toString();
667     }

669     /**
670      * Validate the given language tag according to RFC3066.
671      *
672      * The grammar for a language tag is:
673      * <pre>
674      *   LANGUAGE-TAG   = PRIMARY-SUBTAG ("-" SUBTAG)*
675      *   PRIMARY-SUBTAG = 1-8(ALPHA)
676      *   SUBTAG         = 1-8(ALPHA/DIGIT)
677      * </pre>
678      *
679      * @param language the language to validate.
680      * @throws ParseException if the language is syntactically invalid.
681      */
682     public static void validateLanguage(final String language)
683             throws ParseException {
684         if (language == null || language.length() == 0) {
685             throw new ParseException("Language tag must not be empty", 0);
686         } else if (language.indexOf(" ") != -1) {
687             throw new ParseException("Space character not allowed in "
688                     + "language tag", language.indexOf(" "));
689         } else if (language.startsWith("-") || language.endsWith("-")) {
690             throw new ParseException("Language tag cannot start or end with -",
691                     0);
692         } else {
693             validateLanguageSubtags(language.split("-"));
694         }
695     }
696 
697     /**
698      * Validate each subtag of a language tag.
699      *
700      * @param subtags one or more subtags.
701      * @throws ParseException if any of the subtags are syntactically invalid.
702      */
703     private static void validateLanguageSubtags(final String[] subtags)
704             throws ParseException {
705         for (int i = 0; i < subtags.length; i++) {
706             if (subtags[i].length() < 1
707                     || subtags[i].length() > SUBTAG_MAXLEN) {
708                 throw new ParseException("Language subtags must be "
709                         + "1-" + SUBTAG_MAXLEN + " characters long", 0);
710             }
711             validateLanguageSubtagChars(subtags[i], i == 0);
712         }
713     }

715     /**
716      * Validate each character of a language subtag.
717      *
718      * @param subtag the subtag to validate.
719      * @param isFirst whether the subtag is the first one in the language tag.
720      * @throws ParseException if any of the characters are invalid.
721      */
722     private static void validateLanguageSubtagChars(final String subtag,
723                                                     final boolean isFirst)
724             throws ParseException {
725         for (int i = 0; i < subtag.length(); i++) {
726             char c = subtag.charAt(i);
