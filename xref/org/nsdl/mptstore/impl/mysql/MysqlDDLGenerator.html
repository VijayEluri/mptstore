4   import java.util.ArrayList;
5   import java.util.List;

7   import org.nsdl.mptstore.core.AbstractDDLGenerator;

9   /** DDLGenerator that works with Mysql.
10   * <p>
11   *   The map table DDL defines pKey as a <code>AUTO INCREMENT</code> value
12   *   and p as a <code>VARCHAR(N)</code> value, with indexes on each column.
13   *   The default length N is 255, but this can be modified by setting a system
14   *   property or as a constructor argument.
15   * </p>
16   * <p>
17   *   The relationship table DDL defines s and o
18   *   as <code>VARCHAR(N)</code> values, with indexes on each column.
19   *   The default length N is 255, but this can be modified by setting a system
20   *   property or as a constructor argument.
21   * </p>
22   * <p>
23   *   The following system properties, if set, modify the default behaviour:
24   *   <dl>
25   *    <dt>mptstore.mysql.length</dt>
26   *    <dd>
27   *     Use this to set the specified varchar length when creating map or
28   *     predicate tables.  Default is 255, but mysql versions greater than
29   *     5.0.3 can accept up to 65536.
30   *     Example: <code>-Dmptstore.mysql.length=1024</code>
31   *    </dd>
32   *
33   *    <dt>mptstore.mysql.engine</dt>
34   *    <dd>
35   *     Use this to set the default storage engine for newly created tables.
36   *     By default, this is unspecified, so mysql is free to choose based on
37   *     its own policy.
38   *     Example: <code>-Dmptstore.mysql.engine=innodb</code>
39   *    </dd>
40   *   </dl>
41   * </p>
42   *
43   * @author birkland
44   */
45  public class MysqlDDLGenerator
46          extends AbstractDDLGenerator {
47  
48      private static final String DEFAULT_VARCHAR_LENGTH = "255";
49  
50      /** Property name for specifying default varchar length. */
51      public static final String PROP_TEXT_LENGTH = "mptstore.mysql.length";
52  
53      /** Property name for specifying mysql storage engine. */
54      public static final String PROP_STORAGE_ENGINE = "mptstore.mysql.engine";
55  
56      private final int _length;
57  
58      /**
59       * Construct a MysqlDDLGenerator that uses the default maximum length (255)
60       * for varchar columns that store URIs and literals.
61       */
62      public MysqlDDLGenerator() {
63          _length =
64                  new Integer(System.getProperty(PROP_TEXT_LENGTH,
65                                                 DEFAULT_VARCHAR_LENGTH));
66      }
67  
68      /**
69       * Construct a DerbyDDLGenerator that uses a given maximum length
70       * for varchar columns that store URIs and literals.  For mysql versions
71       * newer than 5.0.3, the maximum is 65535.
72       *
73       * @param length default varchar length.
74       */
75      public MysqlDDLGenerator(final int length) {
76          _length = length;
77      }
78  
79      /** {@inheritDoc} */
80      public List<String> getCreateMapTableDDL(final String table) {
81  
82          List<String> cmds = new ArrayList<String>();
83          cmds.add("CREATE TABLE " + table + " (\n"
84                  + "  pKey INT UNIQUE NOT NULL AUTO_INCREMENT,\n"
85                  + "  p VARCHAR(" + _length + ") UNIQUE NOT NULL\n"
86                  + ") " + getEngine());
87          return cmds;
88      }
89  
90      /** {@inheritDoc} */
91      public List<String> getCreateSOTableDDL(final String table) {
92  
93          List<String> cmds = new ArrayList<String>();
94  
95          cmds.add("CREATE TABLE " + table + " (\n"
96                  + "  s VARCHAR(" + _length + ") NOT NULL,\n"
97                  + "  o VARCHAR(" + _length + ") NOT NULL,\n"
98                  + "  INDEX " + table + "_s (s),\n"
99                  + "  INDEX " + table + "_o (o)\n"
100                 + ")" + getEngine());
101         return cmds;
102     }
103 
104     private String getEngine() {
105         String engine = "";
106         if (System.getProperty(PROP_STORAGE_ENGINE) != null) {
107             engine = "ENGINE " + System.getProperty(PROP_STORAGE_ENGINE);
108         }
109 
110         return engine;
111     }
