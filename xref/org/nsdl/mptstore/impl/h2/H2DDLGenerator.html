7   /**
8    * A <code>DDLGenerator</code> that works with H2.
9    * <p>
10   *   The map table DDL defines pKey as a <code>INT AUTO_INCREMENT</code> value
11   *   and p as an unbound <code>VARCHAR</code> value, with indexes on each
12   *   column.
13   * </p>
14   * <p>
15   *   The relationship table DDL defines s and o
16   *   as <code>VARCHAR</code> values, with indexes on each column.
17   * </p>
18   * <p>
19   *   In H2, there is no pre-defined limit on the length of a
20   *   <code>VARCHAR</code>, unless one is provided.  <code>VARCHAR</code>s
21   *   must, however, be able to fit entirely in memory.  As a result, this
22   *   DDLGenerator does not specify an artificial limit to their length.
23   * </p>
24   *
25   * @author birkland
26   */
27  public class H2DDLGenerator extends AbstractDDLGenerator {
28  
29      /** Create an H2 DDLGenerator with no length limit textual values. */
30      public H2DDLGenerator() {
31      }
32  
33      /** {@inheritDoc} */
34      public List<String> getCreateMapTableDDL(final String table) {
35  
36          List<String> cmds = new ArrayList<String>();
37  
38          cmds.add("CREATE TABLE " + table + " (\n"
39                 + "  pKey INT AUTO_INCREMENT,\n"
40                 + "  p VARCHAR NOT NULL\n"
41                 + ")");
42          cmds.add("CREATE INDEX " + table + "_pKey "
43                 + " on " + table + " (pKey)");
44          cmds.add("CREATE INDEX " + table + "_p "
45                 + " on " + table + " (p)");
46  
47          return cmds;
48      }
49  
50      /** {@inheritDoc} */
51      public List<String> getCreateSOTableDDL(final String table) {
52          List<String> cmds = new ArrayList<String>();
53  
54          cmds.add("CREATE TABLE " + table + " (\n"
55                 + "  s VARCHAR NOT NULL,\n"
56                 + "  o VARCHAR NOT NULL\n"
57                 + ")");
58          cmds.add("CREATE INDEX " + table + "_s "
59                 + " on " + table + " (s)");
60          cmds.add("CREATE INDEX " + table + "_o "
61                 + " on " + table + " (o)");
