8   /**
9    * A <code>DDLGenerator</code> that works with Derby.
10   * <p>
11   *   The map table DDL defines pKey as <code>INT NOT NULL GENERATED ALWAYS
12   *   AS IDENTITY</code>, and p as a VARCHAR, with indexes on each column.
13   * </p>
14   * <p>
15   *   The relationship table DDL defines s and o as <code>VARCHAR</code>
16   *   values, with indexes on each column.
17   * </p>
18   * <p>
19   *   The default VARCHAR length is 512, but this can be overridden by using
20   *   the appropriate constructor.  Note: Derby's limit for VARCHAR length
21   *   is actually 32,672 characters, but because these values are indexed,
22   *   the actual maximum possible here is lower and dependent on the page
23   *   size that Derby has been configured to use. See the <i>Derby Reference
24   *   Manual</i> for more information.
25   * </p>
26   *
27   * @see <a href="http://db.apache.org/derby/docs/10.1/ref/">
28   *        Derby Reference Manual</a>
29   * @author cwilper@cs.cornell.edu
30   */
31  public class DerbyDDLGenerator extends AbstractDDLGenerator {
32  
33      private static final int DEFAULT_VARCHAR_LENGTH = 512;
34  
35      private final int _varcharLength;
36  
37      /**
38       * Construct a DerbyDDLGenerator that uses the default maximum length (512)
39       * for varchar columns that store URIs and literals.
40       */
41      public DerbyDDLGenerator() {
42          _varcharLength = DEFAULT_VARCHAR_LENGTH;
43      }
44  
45      /**
46       * Construct a DerbyDDLGenerator that uses the specified maximum length
47       * for varchar columns that store URIs and literals.
48       *
49       * @param varcharLength the length for all VARCHAR columns.
50       */
51      public DerbyDDLGenerator(final int varcharLength) {
52          _varcharLength = varcharLength;
53      }
54  
55      /** {@inheritDoc} */
56      public List<String> getCreateMapTableDDL(final String table) {
57  
58          List<String> cmds = new ArrayList<String>();
59  
60          cmds.add("CREATE TABLE " + table + " (\n"
61                 + "  pKey INT NOT NULL GENERATED ALWAYS AS IDENTITY,\n"
62                 + "  p VARCHAR(" + _varcharLength + ") NOT NULL\n"
63                 + ")");
64          cmds.add("CREATE INDEX " + table + "_pKey "
65                 + " on " + table + " (pKey)");
66          cmds.add("CREATE INDEX " + table + "_p "
67                 + " on " + table + " (p)");
68  
69          return cmds;
70      }
71  
72      /** {@inheritDoc} */
73      public List<String> getCreateSOTableDDL(final String table) {
74  
75          List<String> cmds = new ArrayList<String>();
76  
77          cmds.add("CREATE TABLE " + table + " (\n"
78                 + "  s VARCHAR(" + _varcharLength + ") NOT NULL,\n"
79                 + "  o VARCHAR(" + _varcharLength + ") NOT NULL\n"
80                 + ")");
81          cmds.add("CREATE INDEX " + table + "_s "
82                 + " on " + table + " (s)");
83          cmds.add("CREATE INDEX " + table + "_o "
84                 + " on " + table + " (o)");
85  
86          return cmds;
87      }
